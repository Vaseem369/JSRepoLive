Topic 1 Function Declarations — function name(params) { body }

Visual reference 

Syntax:   function name(params) { ...body }
To call:   name(arguments)

Term          Means
Parameter     Placeholder in the definition
Argument      Actual value passed when calling
Call / Invoke Run the function: name(args)

function greet(name) {
  console.log("Hello, " + name + "!");
}

greet("Priya");
greet("Aarav");

function add(a, b) {

console.log(add(10, 20));

console.log(add(add(1, 2), add(3, 4)));

                        Topic 3
                    

                    
                        Function Expressions
                    

                

                


                    

                        A function is a value — you can assign it to a variable
                        (function expression).
                    


                    

                        Example
                    


                    
const greet = function(name) {
  return "Hello, " + name;
};

console.log(greet("Priya"));

                        Topic 4
                    

                    
                        Arrow Functions (ES6)
                    

                

                


                    

                        Visual reference
                    


                    
Syntax:   (params) => { body }

Shortcut                        When
Drop parens around 1 param      x => x * 2
Drop braces & return          (a, b) => a + b   (single expression)
Empty parens for zero params    () => "hi"


                    

                        Examples
                    


                    
const add = (a, b) => {
  return a + b;
};

const add2 = (a, b) => a + b;

const square = x => x * x;

const greet = () => "Hello!";

console.log(add(2, 3));
console.log(square(4));
console.log(greet());

console.log(add(2, 3));

                        Topic 5
                    

                    
                        Parameters & Defaults
                    

                

                


                    

                        Default parameters apply when the argument is
                        undefined
                        — not when you pass
                        null,
                        0,
                        or
                        ""
                        (unless you handle those separately).
                    


                    

                        Examples
                    


                    
function greet(name = "Guest") {
  return "Hello, " + name;
}

console.log(greet("Anaya"));
console.log(greet());

const calculatePrice = (price, taxRate = 0.18, discount = 0) => {
  return price * (1 + taxRate) - discount;
};

console.log(calculatePrice(1000));
console.log(calculatePrice(1000, 0.18, 100));


                    

                        Follow along
                    


                    
const greet = (n = "Guest") => "Hello, " + n;
greet("Anaya")
→
your answer
greet()
→
your answer
greet(null)
→
your answer

                        Topic 6
                    

                    
                        Scope: Global vs Local
                    

                

                


                    

                        Global: declared outside any function. Local: declared
                        inside a function. Inner can read outer; locals do not
                        leak out.
                    


                    

                        Example
                    


                    
const globalName = "Priya";

function showName() {
  const localName = "Aarav";
  console.log(globalName);
  console.log(localName);
}

showName();
console.log(globalName);
console.log(localName);   // ReferenceError


                    

                        Reference answers (console order, then error)
                    


                    


                


            


            

            


                

                    
                        Topic 7
                    

                    
                        Pure Functions (intro)
                    

                

                


                    

                        Pure:
                        same inputs → same output; no side effects on the
                        outside world.
                        Impure:
                        reads or writes outer state, I/O, randomness, etc.
                    


                    

                        Example
                    


                    
const add = (a, b) => a + b;

let counter = 0;
function impureAdd(n) {
  counter += n;
  return counter;
}
